let cart = []; let total = 0; function addToCart(name, price) { cart.push({ name, price }); total += price; updateCart(); } function updateCart() { const cartItems = document.getElementById('cart-items'); const cartTotal = document.getElementById('cart-total'); cartItems.innerHTML = ''; cart.forEach(item => { const li = document.createElement('li'); li.textContent = `${item.name} - $${item.price.toFixed(2)}`; cartItems.appendChild(li); }); cartTotal.textContent = total.toFixed(2); } function checkout() { alert(`Total amount: $${total.toFixed(2)}\nThank you for your purchase!`); cart = []; total = 0; updateCart(); }
